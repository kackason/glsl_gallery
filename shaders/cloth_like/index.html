#ifdef GL_ES precision mediump float; #endif uniform vec2 u_resolution; uniform float u_time; float plot(vec2 st, float f){ return smoothstep( f-0.001, f, st.y) - smoothstep( f, f+0.001, st.y); } void main(){ vec2 uv = gl_FragCoord.xy / u_resolution.xy; vec3 color = vec3(1.0); vec2 p = vec2(0.0); vec2 q = vec2(0.0); for(float i=0.0;i<5.0;i+=0.01){ p.x = 0.0; p.y = 0.01*p.y*p.y +0.2*sin(2.0*i + u_time) + 0.1*i; q.x = 1.0; q.y = 0.01*q.y*q.y +0.5*sin(3.0*i +u_time) + 0.15*i; float a = (q.y - p.y)/(q.x - p.x); float func = a*(uv.x - q.x) + q.y; float pct = plot(uv, func); color = (1.0-pct)*color + pct*vec3(0.309, q.x, q.y); } gl_FragColor = vec4(color, 1.0); }
